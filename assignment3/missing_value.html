三、简单随机插补

random.impute <- function(dose) {
    imp <- dose[!is.na(dose)]
    weiyi <- unique(imp)  #确定唯一值进行随机插补
    dose[is.na(dose)] <- sample(weiyi, 1)
    return(dose)
}
earnings.impute = random.impute(wave$earnings)

七、迭代回归插补

iteration = with(wave, data.frame(interest, earnings, male, over65, white, immig, 
    educ_r, workmos, workhrs.top, any.ssi, any.welfare, any.charity))
iteration$interest[interest < 0] <- NA  #将wave中的变量列interest中的负数设置为NA
iteration$icol = ifelse(complete.cases(iteration$interest), "notNA", "imputation")
iteration$ecol = ifelse(complete.cases(iteration$earnings), "notNA", "imputation")

## 可以从summary中看出有很多离群点
iteration$interest <- iteration$interest/100
iteration$interest[iteration$interest > 100] <- 100
iteration$earnings[iteration$earnings > 100] <- 100
summary(iteration$interest)
##    Min. 1st Qu.  Median    Mean 3rd Qu.    Max.    NA's 
##   0.000   0.000   0.000   4.344   0.000 100.000     276
summary(iteration$earnings)
##    Min. 1st Qu.  Median    Mean 3rd Qu.    Max.    NA's 
##    0.00    5.00   28.00   37.89   65.00  100.00     251
interest_lost <- which(is.na(iteration$interest))
earnings_lost <- which(is.na(iteration$earnings))
## 由于interest量纲较大，会影响回归效果，先将两变量平方

iteration$interest = random.impute(iteration$interest)
iteration$earnings = random.impute(iteration$earnings)

bool = TRUE
cishu = 0
while (bool) {
    lm_interest <- lm(interest ~ earnings + male + over65 + white + immig + 
        educ_r + workmos + workhrs.top + any.ssi + any.welfare + any.charity, 
        data = iteration)
    lm_earnings <- lm(earnings ~ interest + male + over65 + white + immig + 
        educ_r + workmos + workhrs.top + any.ssi + any.welfare + any.charity, 
        data = iteration)
    interest_std <- sqrt(mean(lm_interest$residuals^2))
    earnings_std <- sqrt(mean(lm_earnings$residuals^2))
    
    interest_all <- iteration$interest
    earnings_all <- iteration$earnings  #上一步的迭代后的变量值
    
    interest_pre = as.vector(predict(lm_interest, newdata = iteration[interest_lost, 
        ]))
    earnings_pre = as.vector(predict(lm_earnings, newdata = iteration[earnings_lost, 
        ]))  #储存此次迭代值
    pre_index = 1
    for (i in interest_lost) {
        iteration$interest[i] = rnorm(1, mean = interest_pre[pre_index], sd = interest_std)
        pre_index = pre_index + 1
    }
    iteration$interest[iteration$interest < 0] <- 0  #将插补后interest小于0的值全部赋值0
    pre_index = 1
    for (i in earnings_lost) {
        iteration$earnings[i] = rnorm(1, mean = earnings_pre[pre_index], sd = earnings_std)
        pre_index = pre_index + 1
    }
    iteration$earnings[iteration$earnings < 0] <- 0  #将插补后earnings小于0的值全部赋值0
    interest_norm = sqrt(sum((interest_all - iteration$interest)^2))
    earnings_norm = sqrt(sum((earnings_all - iteration$earnings)^2))
    if (interest_norm < 500 && earnings_norm < 500) {
        bool = FALSE
    }
    cishu = cishu + 1
}
cishu  #显示迭代次数 
## [1] 9
