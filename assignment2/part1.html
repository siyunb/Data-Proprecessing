make_datetime_100 <- function(year, month, day, time) {
    make_datetime(year, month, day, time%/%100, time%%100)
    # 对time进行操作取整取余化为标准时间数据
}

result_4<-
  inner_join(flights_df,airports_df,c('dest'='faa')) %>%    
  #内部合并flights_df与airports_df只合并交集
  select(dest,arr_delay)%>%
   na.omit()%>%    #去除空值
  dplyr::mutate(delay_bool=as.integer((arr_delay>0)*1))%>% 
  #计算平均延误时间所以将提前到达和未延误的行去除掉
  dplyr::mutate(delay_time=arr_delay*delay_bool)%>% 
  #计算所有的航班延误时间，提前和未延误的记为零
  group_by(dest)%>%
  summarize(mean_delay=mean(delay_time))

5、研究飞机的使用寿命与延误情况之间关系

计算飞机的使用寿命和延误情况，不能单纯的计算飞机的延误时间总和，因为各个飞机的执行任务的次数不同，所以导致执行任务多的飞机总时间较长，所以应该对飞机的飞行平均的到达延误时间进行对比。

fli_pla_df<-
  inner_join(flights_df,planes_df,c('tailnum'='tailnum')) %>%
  select(arr_delay,tailnum,year.x,year.y)%>%
  na.omit()%>%
  dplyr::mutate(delay_bool=as.integer((arr_delay>0)*1))%>%
  dplyr::mutate(delay_time=arr_delay*delay_bool,use_year=year.x-year.y)%>%
  group_by(tailnum)%>%
  summarize(mean_delay=mean(delay_time),use_year=max(use_year))%>%   #计算平均每一次的delay
  arrange(desc(use_year))
cor(fli_pla_df$use_year,fli_pla_df$mean_delay) 
## [1] -0.02159948
#计算相关系数,相关性很小，甚至成负相关，并无明显关系
ggplot(fli_pla_df,aes(x=use_year,y=mean_delay))+geom_point()+ggtitle('不同使用寿命的平均延误时间') +theme(plot.title = element_text(hjust = 0.5, family="myFont", size=18,color="black"),panel.background=element_rect(fill='aliceblue',color='black'))
